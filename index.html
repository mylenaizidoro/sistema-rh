async function enviarVaga() { const btn = document.getElementById('btn-salvar'); const originalText = btn.innerText; // Captura os dados exatamente como o Apps Script espera const dados = { cargo: document.getElementById('vaga-cargo').value, setor: document.getElementById('vaga-setor').value, gestor: document.getElementById('vaga-gestor').value, salario: document.getElementById('vaga-salario').value, contrato: document.getElementById('vaga-contrato').value, turno: "Integral", // Valor fixo conforme seu modal horario: document.getElementById('vaga-horario').value }; // Validação simples if(!dados.cargo || !dados.setor) return alert("Preencha Cargo e Setor!"); btn.innerText = "Enviando..."; btn.disabled = true; try { // O modo 'no-cors' é usado para evitar bloqueios de segurança do navegador ao enviar para o Google await fetch(urlAPI, { method: 'POST', mode: 'no-cors', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(dados) }); alert("Vaga enviada! (Aguarde alguns segundos para a planilha atualizar)"); fecharModal(); document.getElementById('form-vaga').reset(); // Recarrega os dados da planilha para mostrar a nova vaga na tabela setTimeout(carregarDados, 3000); } catch (e) { alert("Erro ao conectar com a API: " + e.message); } finally { btn.innerText = originalText; btn.disabled = false; } }
